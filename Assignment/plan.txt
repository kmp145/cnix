loadSettings(){
	open the file
	if the file exits{
		if the file is not readable{
			display error message
			log error
			terminate program
		}
		while not end of file{
			read a line
			if the line doesnt start with #{
				if it begins with username{
					assign to settings
				}
				... and so on for rest of settings
				else{
					display error message
					log error
					terminate program
				}
			}	
		}
	}	
	else{
		create the conf file
		assign the default settings
		log the event
	}
	return settings
}
